%union{
	double real;
	int id;
}

%token  <id> INTEIRO
%type   <id>    exp


%right   '='
%left    '-'  '+'
%left    '*'  '/'
%left     NEG
%right  '^'
-------------------------------SEÇÃO DE REGRAS GRAMATICAIS-------------------


%%
input : |input line;
line : '\n' {printf("Entrada de linha em branco");} 
	|exp '\n' {printf("%s\n",$1);} | error '\n' {yyerror;} ;

exp : NUM{$$ = $1;} 
 
     | exp '+' exp  {$$ = $1 + $3;}
     | exp '-' exp  {$$ = $1 - $3;}
     | exp '*' exp  {$$ = $1 * $3;} 
     | exp '/' exp  {$$ = $1 / $3;}
     | '-' exp %prec NEG  {$$ = -$2;} 
     | '(' exp ')' { $$=$2;}
